    <script>
        var map = new naver.maps.Map('map', {
            center: new naver.maps.LatLng(37.5665, 126.9780), // 서울 중심 좌표
            zoom: 7
        });

        var locations = [
            { lat: 37.5656, lng: 126.9769, title: "서울역 수퍼차저", address: "서울특별시 중구 한강대로 405", floor: "B2층", parking_fee: "30분 무료, 이후 10분당 1,000원" },
            { lat: 35.1709, lng: 129.1308, title: "부산 센텀 수퍼차저", address: "부산 해운대구 센텀동로 35", floor: "3층", parking_fee: "1시간 무료, 이후 15분당 500원" },
            { lat: 36.3504, lng: 127.3845, title: "대전 수퍼차저", address: "대전광역시 서구 둔산중로 50", floor: "1층", parking_fee: "20분 무료, 이후 10분당 800원" }
            // 🛠️ 여기에 추가적인 수퍼차저 데이터를 입력 가능!
        ];

        locations.forEach(function (location) {
            var marker = new naver.maps.Marker({
                position: new naver.maps.LatLng(location.lat, location.lng),
                map: map,
                title: location.title
            });

            var infoWindow = new naver.maps.InfoWindow({
                content: `<div style="padding:10px; font-size:14px;">
                            <strong>${location.title}</strong><br>
                            📍 <b>주소:</b> ${location.address}<br>
                            🏬 <b>위치:</b> ${location.floor}<br>
                            💰 <b>주차 요금:</b> ${location.parking_fee}
                          </div>`
            });

            naver.maps.Event.addListener(marker, "click", function() {
                infoWindow.open(map, marker);
            });
        });
    </script>
